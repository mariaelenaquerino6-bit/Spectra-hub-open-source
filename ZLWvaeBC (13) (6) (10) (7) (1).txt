local function BringBus(target)
    local LocalPlayer = game.Players.LocalPlayer
    local RS = game:GetService("ReplicatedStorage")
    local RunService = game:GetService("RunService")

    if not (target and target.Character and LocalPlayer.Character) then return end
    
    local char = LocalPlayer.Character
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    
    if hrp and hum then
        local originalPos = hrp.CFrame
        
        -- Spawn do √înibus (M√©todo Fling Original)
        pcall(function()
            hrp.Anchored = true
            hrp.CFrame = CFrame.new(1181.83, 76.08, -1158.83)
            task.wait(0.2)
            hrp.Velocity, hrp.RotVelocity = Vector3.new(0,0,0), Vector3.new(0,0,0)
            hrp.Anchored = false
            hum:ChangeState(Enum.HumanoidStateType.GettingUp)
        end)
 
        task.wait(0.5)
        pcall(function() RS:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer("DeleteAllVehicles") end)
        pcall(function() RS:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer("PickingCar", "SchoolBus") end)
 
        task.wait(1)
        
        local bus = workspace:FindFirstChild("Vehicles") and workspace.Vehicles:FindFirstChild(LocalPlayer.Name .. "Car")
        if bus then
            -- Sentar no √¥nibus
            pcall(function()
                hrp.Anchored = true
                hrp.CFrame = CFrame.new(1171.15, 79.45, -1166.2)
                task.wait(0.2)
                hrp.Velocity, hrp.RotVelocity = Vector3.new(0,0,0), Vector3.new(0,0,0)
                hrp.Anchored = false
                hum:ChangeState(Enum.HumanoidStateType.Seated)
            end)
            
            task.wait(0.5)
            if not hum.Sit then return end
            
            -- Configura√ß√£o do √¥nibus
            for _, p in pairs(bus:GetDescendants()) do
                if p:IsA("BasePart") then
                    p.CanCollide = false
                    pcall(function() p:SetNetworkOwner(nil) end)
                end
            end
            
            -- Persegui√ß√£o com Giro de Fling
            local running = true
            local connection
            connection = RunService.Heartbeat:Connect(function()
                if not running or not (target.Character and target.Character:FindFirstChild("HumanoidRootPart")) then
                    if connection then connection:Disconnect() end
                    return
                end
                
                local tHRP = target.Character.HumanoidRootPart
                local tHum = target.Character:FindFirstChildOfClass("Humanoid")
                
                -- Giro do Fling Original
                pcall(function()
                    bus:PivotTo(CFrame.new(tHRP.Position + Vector3.new(math.random(-10,10),0,math.random(-10,10))) * CFrame.Angles(
                        math.rad(game.Workspace.DistributedGameTime*12000),
                        math.rad(game.Workspace.DistributedGameTime*15000),
                        math.rad(game.Workspace.DistributedGameTime*18000)
                    ))
                end)
                
                -- Verificar se o alvo sentou
                local targetSat = false
                for _, seat in pairs(bus:GetDescendants()) do
                    if (seat:IsA("Seat") or seat:IsA("VehicleSeat")) and seat.Name ~= "VehicleSeat" and seat.Occupant == tHum then
                        targetSat = true
                        break
                    end
                end
                
                -- BRING (Traz para a sua posi√ß√£o atual)
                if targetSat then
                    running = false
                    connection:Disconnect()
                    
                    -- Teleporta o √¥nibus para a sua frente (2 studs √† frente)
                    local myCurrentPos = hrp.CFrame
                    pcall(function() 
                        bus:PivotTo(myCurrentPos * CFrame.new(0, 0, -2)) 
                    end)
                    
                    -- Pequena pausa para garantir que o player viajou junto com o √¥nibus
                    task.wait(0.1)
                    
                    -- Deleta √¥nibus e solta o player na sua frente
                    pcall(function() RS:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer("DeleteAllVehicles") end)
                    
                    -- Garante que voc√™ n√£o fique preso
                    pcall(function()
                        hum:ChangeState(Enum.HumanoidStateType.GettingUp)
                    end)
                end
            end)
            
            task.delay(15, function() if running then running = false if connection then connection:Disconnect() end end end)
        end
    end
end

-- Fun√ß√£o para Kill Couch
local function KillCouch(targetPlayer)
    if _G.IsTrapping then return end
    _G.IsTrapping = true
    
    local playerChar = LPlayer.Character
    local playerHRP = playerChar and playerChar:FindFirstChild("HumanoidRootPart")
    local playerHumanoid = playerChar and playerChar:FindFirstChildOfClass("Humanoid")
    
    if not playerHRP or not playerHumanoid then _G.IsTrapping = false return end
    
    local originalPos = playerHRP.CFrame -- Salva posi√ß√£o original

    -- Seleciona quem vai morrer
    local targets = {}
    local t = Players:FindFirstChild(_G.SelectedPlayer)
    if t then table.insert(targets, t) end

    -- Inicia o Massacre
    for _, targetPlayer in ipairs(targets) do
        if not _G.IsTrapping then break end
        
        if targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
            local targetHRP = targetPlayer.Character.HumanoidRootPart
            local targetHum = targetPlayer.Character:FindFirstChildOfClass("Humanoid")
            
            if targetHum and not targetHum.Sit and targetHum.Health > 0 then
                -- Pega o Sof√°
                pcall(function()
                    RS.RE["1Clea1rTool1s"]:FireServer("ClearAllTools")
                    task.wait(0.2)
                    RS.RE["1Too1l"]:InvokeServer("PickingTools", "Couch")
                end)
                
                task.wait(0.3)
                local couch = LPlayer.Backpack:FindFirstChild("Couch")
                if couch then couch.Parent = playerChar end
                
                task.wait(0.1)
                game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.F, false, game)
                task.wait(0.25)
                
                -- F√≠sica de Fling
                workspace.FallenPartsDestroyHeight = 0/0
                local bodyVel = Instance.new("BodyVelocity")
                bodyVel.Name = "FlingForce"
                bodyVel.Velocity = Vector3.new(900000000, 900000000, 900000000)
                bodyVel.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
                bodyVel.Parent = playerHRP
                
                playerHumanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
                workspace.CurrentCamera.CameraSubject = targetPlayer.Character:FindFirstChild("Head") or targetHRP
                
                local rotation = 0
                local startTime = tick()
                
                -- Loop de ataque no alvo atual
                while _G.IsTrapping and targetPlayer.Parent and targetHum.Health > 0 and not targetHum.Sit do
                    if tick() - startTime > 6 then break end -- 6 segundos por alvo
                    RunService.Heartbeat:Wait()
                    
                    rotation = rotation + 50
                    local targetPos = targetHRP.Position + targetHRP.Velocity / 1.5
                    playerHRP.CFrame = CFrame.new(targetPos) * CFrame.Angles(math.rad(rotation), 0, 0)
                    
                    playerHRP.Velocity = Vector3.new(900000000, 900000000, 900000000)
                    playerHRP.RotVelocity = Vector3.new(900000000, 900000000, 900000000)
                end
                
                -- Limpeza r√°pida para o pr√≥ximo alvo
                bodyVel:Destroy()
                playerHumanoid:UnequipTools()
                pcall(function() RS.RE["1Clea1rTool1s"]:FireServer("ClearAllTools") end)
                
                for _, part in pairs(playerChar:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.Velocity = Vector3.zero
                        part.RotVelocity = Vector3.zero
                    end
                end
            end
        end
    end

    -- Finaliza√ß√£o Total
    _G.IsTrapping = false
    playerHumanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
    playerHRP.Anchored = true
    playerHRP.CFrame = originalPos
    task.wait(0.2)
    playerHRP.Velocity = Vector3.zero
    playerHRP.RotVelocity = Vector3.zero
    playerHRP.Anchored = false
    workspace.CurrentCamera.CameraSubject = playerHumanoid
end

-- [[ FUN√á√ÉO BRING COUCH - SERVER POSITION FIX ]]
local function BringCouch(targetPlayer)
    if _G.IsTrapping then return end
    _G.IsTrapping = true
    
    local playerChar = LPlayer.Character
    local playerHRP = playerChar and playerChar:FindFirstChild("HumanoidRootPart")
    local playerHumanoid = playerChar and playerChar:FindFirstChildOfClass("Humanoid")
    
    if not playerHRP or not playerHumanoid then 
        _G.IsTrapping = false 
        return 
    end
    
    local originalPos = playerHRP.CFrame

    -- 1. Limpeza e Requisi√ß√£o
    pcall(function()
        RS.RE["1Clea1rTool1s"]:FireServer("ClearAllTools")
    end)
    task.wait(0.1)

    pcall(function()
        RS.RE["1Too1l"]:InvokeServer("PickingTools", "Couch")
    end)
    
    local couch = LPlayer.Backpack:WaitForChild("Couch", 3)
    if not couch then _G.IsTrapping = false return end
    
    couch.Parent = playerChar
    task.wait(0.1) -- Tempo para o servidor registrar a ferramenta equipada

    -- 2. Ativa√ß√£o do Sof√°
    game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.F, false, game)
    task.wait(0.2) -- Delay para o sof√° ser instanciado fisicamente
    
    playerHumanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
    
    local targetChar = targetPlayer.Character
    local targetHRP = targetChar and targetChar:FindFirstChild("HumanoidRootPart")
    local targetHum = targetChar and targetChar:FindFirstChildOfClass("Humanoid")
    
    if not targetHRP or not targetHum then _G.IsTrapping = false return end
    
    local rotation = 0
    local startTime = tick()
    local success = false
    
    -- 3. Loop de Captura (Giro)
    while tick() - startTime < 6 and not targetHum.Sit do
        RunService.Heartbeat:Wait()
        rotation = rotation + 50
        local targetPos = targetHRP.Position + targetHRP.Velocity / 1.5
        playerHRP.CFrame = CFrame.new(targetPos) * CFrame.Angles(math.rad(rotation), 0, 0)
        
        if targetHum.Sit then
            success = true
            break
        end
    end
    
    -- 4. O PONTO CHAVE: Retorno e Registro de Posi√ß√£o
    if success then
        -- Teleporta para o destino
        playerHRP.CFrame = originalPos
        
        -- AGUARDA o servidor entender que o sof√° est√° na nova posi√ß√£o.
        -- Se deletar antes desse tempo, o servidor cancela o movimento e volta o player.
        task.wait(0.5) 
        
        -- For√ßamos um pequeno movimento para atualizar o estado de rede do sof√° na nova posi√ß√£o
        playerHRP.CFrame = originalPos * CFrame.new(0, 0.1, 0)
        task.wait(0.1)
        playerHRP.CFrame = originalPos
        
        -- Aguarda o "congelamento" da posi√ß√£o no servidor do alvo
        task.wait(0.4) 
    else
        playerHRP.CFrame = originalPos
    end
    
    -- 5. Dele√ß√£o do Sof√°
    -- Ao deletar aqui, o player cai. Se ele voltar, √© porque o servidor dele 
    -- ainda n√£o tinha recebido os pacotes da nova posi√ß√£o (Lag/Ping).
    pcall(function() 
        RS.RE["1Clea1rTool1s"]:FireServer("ClearAllTools") 
    end)
    
    task.wait(0.2)
    playerHRP.Velocity = Vector3.zero
    playerHRP.Anchored = false
    _G.IsTrapping = false
    playerHumanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
    
    if success then
        Window:Notify({Title = "Bring", Content = "Jogador entregue.", Duration = 3})
    end
end

-- [[ FUN√á√ÉO DE FLING (ADAPTADA DO TEU C√ìDIGO) ]]
local function ExecuteFling(target)
    local player = lp
    local character = player.Character or player.CharacterAdded:Wait()
    local humanoid = character:WaitForChild("Humanoid")
    local backpack = player:WaitForChild("Backpack")
 
    local workspaceCom = workspace:FindFirstChild("WorkspaceCom")
    if not workspaceCom then return end
    local ServerBalls = workspaceCom:WaitForChild("001_SoccerBalls")
 
    -- Fun√ß√£o Interna para pegar a bola
    local function GetBall()
        if not backpack:FindFirstChild("SoccerBall") and not character:FindFirstChild("SoccerBall") then
            game:GetService("ReplicatedStorage").RE:FindFirstChild("1Too1l"):InvokeServer("PickingTools", "SoccerBall")
        end
 
        local tool = backpack:WaitForChild("SoccerBall", 5) or character:FindFirstChild("SoccerBall")
        if not tool then return nil end
 
        tool.Parent = character
        local ballName = "Soccer" .. player.Name
        local foundBall = nil
        local timeout = tick()
 
        repeat 
            task.wait() 
            foundBall = ServerBalls:FindFirstChild(ballName)
        until foundBall or (tick() - timeout > 5)
 
        if tool.Parent == character then tool.Parent = backpack end
        return foundBall
    end
 
    local Ball = ServerBalls:FindFirstChild("Soccer" .. player.Name) or GetBall()
 
    if not Ball then return end
 
    -- F√çSICA DO TEU SCRIPT
    Ball.CanCollide = false
    Ball.Massless = true
    Ball.CustomPhysicalProperties = PhysicalProperties.new(100, 0, 0, 100, 100)
 
    if target and target ~= player then
        local tchar = target.Character
        if tchar and tchar:FindFirstChild("HumanoidRootPart") and tchar:FindFirstChild("Humanoid") then
            local troot = tchar.HumanoidRootPart
            local thum = tchar.Humanoid
 
            for _, v in pairs(Ball:GetChildren()) do
                if v:IsA("BodyVelocity") or v:IsA("BodyForce") or v:IsA("BodyAngularVelocity") then v:Destroy() end
            end
 
            local bv = Instance.new("BodyVelocity")
            bv.Name = "FlingPower"
            bv.Velocity = Vector3.new(9e8, 9e8, 9e8)
            bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
            bv.P = 9e9
            bv.Parent = Ball
            
            Ball.RotVelocity = Vector3.new(10000, 10000, 10000)
            workspace.CurrentCamera.CameraSubject = thum
 
            local connection
            connection = game:GetService("RunService").Heartbeat:Connect(function()
                if not Ball or not Ball.Parent or not troot or not troot.Parent or thum.Health <= 0 then
                    connection:Disconnect()
                    workspace.CurrentCamera.CameraSubject = humanoid
                    return
                end
 
                if troot.Velocity.Magnitude > 0.1 then
                    local Prediction = troot.Velocity * 0.12
                    Ball.CFrame = CFrame.new(troot.Position + Prediction)
                else
                    Ball.CFrame = troot.CFrame
                end
                
                Ball.RotVelocity = Vector3.new(10000, 10000, 10000)
            end)
 
            task.delay(30, function()
                if connection then connection:Disconnect() end
                if workspace.CurrentCamera.CameraSubject == thum then
                    workspace.CurrentCamera.CameraSubject = humanoid
                end
            end)
        end
    end
end

local function ExecuteBoatTeleport()
    local Player = game.Players.LocalPlayer

    local Humanoid = Character:WaitForChild("Humanoid")
    local RootPart = Character:WaitForChild("HumanoidRootPart")

    local PlazaPos = CFrame.new(-26, 4, -23)  -- Local da Pra√ßa
    
    -- 1. Verificar se j√° tem um carro/barco e deletar se necess√°rio (opcional)
    local PCar = Vehicles:FindFirstChild(Player.Name.."Car")

    RootPart.CFrame = SpawnPos
    task.wait(0.5)
    
    -- 3. Spawnar o Barco Militar
    game:GetService("ReplicatedStorage").RE:FindFirstChild("1Ca1r"):FireServer("PickingBoat", "MilitaryBoatFree")
    
    -- 4. Esperar o barco aparecer no Workspace
    local boatTimeout = 0
    repeat
        task.wait(0.1)
        PCar = Vehicles:FindFirstChild(Player.Name.."Car")
        boatTimeout = boatTimeout + 1
    until PCar or boatTimeout > 50
    
    if PCar then
        -- 5. Entrar no assento do piloto
        local Seat = PCar:FindFirstChildWhichIsA("VehicleSeat", true)
        if Seat then
            repeat
                task.wait()
                RootPart.CFrame = Seat.CFrame
            until Humanoid.Sit
            
            -- Pequena pausa para garantir que o jogo entendeu que voc√™ sentou
            task.wait(0.3)
            
            -- 6. Teleportar o Barco inteiro para a Pra√ßa
            -- Movemos o PrimaryPart do carro para levar tudo junto
            if PCar.PrimaryPart then
                PCar:SetPrimaryPartCFrame(PlazaPos)
            else
                -- Fallback caso n√£o tenha PrimaryPart definido
                Seat.CFrame = PlazaPos
            end
            
            task.wait(0.5)
            
            -- 7. Pular do barco (Ejetar)
            Humanoid.Sit = false
            task.wait(0.1)
            Humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
            
            -- 8. Pequeno ajuste de posi√ß√£o final para n√£o bugar no ch√£o
            RootPart.CFrame = PlazaPos + Vector3.new(0, 3, 5)
        end
    end
end

local function SpawnPiratePlaza()
    local character = localPlayer.Character

    for i = 1, 50 do

        local seat = pCar:FindFirstChildWhichIsA("VehicleSeat", true)
        if seat then
            -- 5. Senta no barco
            repeat
                task.wait()
                hrp.CFrame = seat.CFrame
            until hum.Sit
            
            task.wait(0.3)
            
            -- 6. Teleporta o barco inteiro para a Pra√É¬ßa (Sem girar)
            if pCar.PrimaryPart then
                pCar:SetPrimaryPartCFrame(placaPos)
            else
                seat.CFrame = placaPos
            end
            
            task.wait(0.5)
            
            -- 7. Pula do barco
            hum.Sit = false
            task.wait(0.1)
            hum:ChangeState(Enum.HumanoidStateType.Jumping)
            hrp.CFrame = placaPos + Vector3.new(0, 5, 0)
        end
    end

local function ApplyVehicleCollision(vehicle)
    local vehicleTypes = {
        "Shark", "WaveRunner", "SailBoatFree", "SpeedBoatFree",
        "PoliceBoatChangeable", "MilitaryBoatFree", "PirateFree",
        "Jet", "Helicopter"
    }
    
    for _, part in pairs(vehicle:GetDescendants()) do
        if part:IsA("BasePart") then
            local partName = part.Name:lower()
            local isWaterVehicle = false
            
            for _, vType in ipairs(vehicleTypes) do
                if vehicle.Name == vType then
                    isWaterVehicle = true
                    break
                end
            end
            
            part.CanCollide = isWaterVehicle and (
                part:IsA("Seat") or part:IsA("VehicleSeat") or
                partName:find("wheel") or partName:find("tire") or
                partName:find("chassis") or partName:find("base") or
                partName:find("frame") or partName:find("hull") or
                partName:find("propeller") or partName:find("rotor") or
                partName:find("wing")
            ) or (
                part:IsA("Seat") or part:IsA("VehicleSeat") or
                partName:find("wheel") or partName:find("tire") or
                partName:find("chassis") or partName:find("base") or
                partName:find("frame")
            )
        end
    end
end
